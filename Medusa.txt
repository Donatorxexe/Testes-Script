local function hasFunction(name)
    local f = nil
    pcall(function()
        if getgenv and getgenv()[name] then f = getgenv()[name]
        elseif _G and _G[name] then f = _G[name]
        else f = getfenv()[name] end

        pcall(function() f = getfenv(0)[name] end)
    end
    return f
end

    if obj.connections[name] then
        pcall(function() obj.connections[name]:Disconnect() end)

end

    if obj.connections[name] then
        pcall(function() obj.connections[name]:Disconnect() end)
        obj.connections[name] = nil
    end
end

        pcall(function() conn:Disconnect() end)

-- ── Save/Load Config ──
local function saveConfig()
    if not _writefile then return false end
    local ok, err = pcall(function()
        local data = {
            cfg = {},
            keybinds = {},
            theme = {},
            st = {},
        }
        -- Salvar cfg
        for k, v in pairs(cfg) do
            if type(v) == "number" or type(v) == "boolean" or type(v) == "string" then
                data.cfg[k] = v
            end
        end
        -- Salvar keybinds (como nomes)
        for k, v in pairs(keybinds) do
            data.keybinds[k] = v.Name
        end
        -- Salvar cores do tema atual
        for k, v in pairs(C) do
            if typeof(v) == "Color3" then
                data.theme[k] = {math.floor(v.R*255), math.floor(v.G*255), math.floor(v.B*255)}
            end
        end
        -- Salvar estados de toggle
        for k, v in pairs(st) do
            if type(v) == "boolean" then
                data.st[k] = v
            end
        end
        local json = HttpService:JSONEncode(data)
        _writefile(CONFIG_FILE, json)
    end)
    return ok
end

local function loadConfig()
    if not _readfile or not _isfile then return false end
    local ok = pcall(function()
        if not _isfile(CONFIG_FILE) then return end
        local raw = _readfile(CONFIG_FILE)
        local data = HttpService:JSONDecode(raw)

        -- Restaurar cfg
        if data.cfg then
            for k, v in pairs(data.cfg) do
                if cfg[k] ~= nil and type(cfg[k]) == type(v) then
                    cfg[k] = v
                end
            end
        end

        -- Restaurar keybinds
        if data.keybinds then
            for k, v in pairs(data.keybinds) do
                if keybinds[k] and Enum.KeyCode[v] then
                    keybinds[k] = Enum.KeyCode[v]
                end
            end
        end

        -- Restaurar cores
        if data.theme then
            for k, v in pairs(data.theme) do
                if C[k] and type(v) == "table" and #v == 3 then
                    C[k] = Color3.fromRGB(v[1], v[2], v[3])
                end
            end
        end

        -- Restaurar estados (NÃO restauramos toggles perigosos)
        -- Apenas estados visuais
        if data.st then
            local safeKeys = {
                "crosshair", "rainbow", "ghostMode",
            }
            for _, k in ipairs(safeKeys) do
                if data.st[k] ~= nil then
                    st[k] = data.st[k]
                end
            end
        end
    end)
    return ok
end

    -- Método 3: CoreGui direto (fallback)
    if not SP.hidden then
        local ok = pcall(function()
            local test = Instance.new("Folder")
            test.Name = "_MedusaSPTest"
            test.Parent = CoreGui
            test:Destroy()
        end)
        if ok then
            SP.hidden = CoreGui
            SP.method = "CoreGui"
        end
    end

    -- Método 4: protect_gui + CoreGui
    if not SP.hidden and _protectgui then
        SP.hidden = CoreGui
        SP.method = "protect"
    end
end

print("[MEDUSA] StreamProof Method: " .. SP.method)
print("[MEDUSA] Hidden container: " .. tostring(SP.hidden))

-- ── Criar ScreenGui no parent correto ────────────────────────
-- TODAS as GUIs passam por esta função
local function createGui(name, displayOrder, resetOnSpawn)
    local gui = Instance.new("ScreenGui")
    gui.Name = name or "MedusaGui"
    gui.DisplayOrder = displayOrder or 999
    gui.ResetOnSpawn = resetOnSpawn or false
    gui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    gui.IgnoreGuiInset = false

    -- Se StreamProof activo → parent no container oculto
    if SP.active and SP.hidden then
        if SP.method == "protect" and _protectgui then
            pcall(_protectgui, gui)
        end
        gui.Parent = SP.hidden
    else
        gui.Parent = playerGui
    end

    table.insert(SP.allGuis, gui)
    return gui
end

-- ── Mover TODAS as GUIs para o container oculto ─────────────
local function spEnable()
    if not SP.hidden then
        return false, "Sem método disponível"
    end

    for _, gui in ipairs(SP.allGuis) do
        if gui and gui.Parent then
            pcall(function()
                if SP.method == "protect" and _protectgui then
                    _protectgui(gui)
                end
                gui.Parent = SP.hidden
            end)
        end
    end

    SP.active = true
    st.streamProof = true
    print("[MEDUSA] ✅ StreamProof ATIVADO via " .. SP.method)
    return true, SP.method
end

-- ── Mover TODAS as GUIs de volta para PlayerGui ─────────────

    for _, gui in ipairs(SP.allGuis) do
        if gui and gui.Parent then
            pcall(function()
                gui.Parent = playerGui
            end)
        end
    end

    SP.active = false
    st.streamProof = false
    print("[MEDUSA] ❌ StreamProof DESATIVADO")
end

-- ── Ghost Mode Engine ────────────────────────────────────────
-- Quando ativo: menu fica 95% transparente
-- Quando o rato passa por cima: aparece normal
-- Usa polling porque InputChanged nem sempre deteta hover em frames

local ghostTargets = {}  -- {frame, origTransparency}

local function registerGhostTarget(frame, origTransp)
    table.insert(ghostTargets, {frame = frame, orig = origTransp or 0})
end

local function ghostApply(hovering)
    SP.ghostHover = hovering
    for _, data in ipairs(ghostTargets) do
        if data.frame and data.frame.Parent then
            pcall(function()
                if SP.ghostMode and not hovering then
                    data.frame.BackgroundTransparency = 0.95
                else
                    data.frame.BackgroundTransparency = data.orig
                end
            end)
        end
    end
end

local function ghostEnable()
    SP.ghostMode = true
    st.ghostMode = true
    ghostApply(false)
    print("[MEDUSA] 👻 Ghost Mode ATIVADO")
end

local function ghostDisable()
    SP.ghostMode = false
    st.ghostMode = false
    ghostApply(true)  -- restaura tudo
    print("[MEDUSA] 👻 Ghost Mode DESATIVADO")
end

-- ── mkToggle ─────────────────────────────────────────────────
-- Estilos: "Dot" | "Check" | "Fill" | "Slash"
-- Retorna: {frame, knob, update(on), state}

local function mkToggle(parent, labelText, icon, initState, x, y, callback, style)
    style = style or cfg.gui.toggleStyle
    local W = cfg.gui.toggleW
    local H = cfg.gui.toggleH

    -- Container
    local row = Instance.new("Frame")
    row.Name = "Toggle_" .. labelText
    row.Size = UDim2.new(1, -28, 0, math.max(H + 4, 24))
    row.Position = UDim2.new(0, 14, 0, y)
    row.BackgroundTransparency = 1
    row.Parent = parent

    -- Icon + Label
    local lbl = Instance.new("TextLabel")
    lbl.Size = UDim2.new(1, -(W + 10), 1, 0)
    lbl.BackgroundTransparency = 1
    lbl.Font = Enum.Font.GothamMedium
    lbl.TextSize = cfg.gui.fontSize
    lbl.TextColor3 = C.text
    lbl.TextXAlignment = Enum.TextXAlignment.Left
    lbl.Text = (icon and icon .. "  " or "") .. labelText
    lbl.Parent = row

    -- Toggle track
    local track = Instance.new("Frame")
    track.Name = "Track"
    track.Size = UDim2.new(0, W, 0, H)
    track.Position = UDim2.new(1, -W, 0.5, -H/2)
    track.AnchorPoint = Vector2.new(0, 0)
    track.BorderSizePixel = 0
    track.Parent = row

    local trackCorner = Instance.new("UICorner", track)

    -- Indicador/Knob (depende do estilo)
    local knob = Instance.new("Frame")
    knob.Name = "Knob"
    knob.BorderSizePixel = 0
    knob.Parent = track
    local knobCorner = Instance.new("UICorner", knob)

    -- Texto interno para Check/Slash
    local checkLbl = Instance.new("TextLabel")
    checkLbl.Size = UDim2.new(1, 0, 1, 0)
    checkLbl.BackgroundTransparency = 1
    checkLbl.Font = Enum.Font.GothamBold
    checkLbl.TextColor3 = Color3.new(1, 1, 1)
    checkLbl.Parent = knob

    -- Estado
    local state = {on = initState}

    -- Função de update visual
    local function updateVisual(on, animate)
        state.on = on

        if style == "Dot" then
            -- Estilo clássico: knob desliza esquerda/direita
            trackCorner.CornerRadius = UDim.new(1, 0)
            knobCorner.CornerRadius = UDim.new(1, 0)
            knob.Size = UDim2.new(0, H - 4, 0, H - 4)
            knob.AnchorPoint = Vector2.new(0, 0.5)
            checkLbl.Visible = false

            local targetPos = on
                and UDim2.new(1, -(H - 2), 0.5, 0)
                or  UDim2.new(0, 2, 0.5, 0)
            local targetBG = on and C.accent or C.border

            if animate then
                TweenService:Create(knob, TweenInfo.new(0.2, Enum.EasingStyle.Back), {
                    Position = targetPos
                }):Play()
                TweenService:Create(track, TweenInfo.new(0.15), {
                    BackgroundColor3 = targetBG
                }):Play()
            else
                knob.Position = targetPos
                track.BackgroundColor3 = targetBG
            end
            knob.BackgroundColor3 = Color3.new(1, 1, 1)

        elseif style == "Check" then
            -- Estilo checkbox: mostra ✓ quando ON
            trackCorner.CornerRadius = UDim.new(0, cfg.gui.cornerRadius)
            knobCorner.CornerRadius = UDim.new(0, cfg.gui.cornerRadius)
            knob.Size = UDim2.new(1, -4, 1, -4)
            knob.Position = UDim2.new(0, 2, 0, 2)
            knob.AnchorPoint = Vector2.new(0, 0)
            checkLbl.Visible = true
            checkLbl.TextSize = H - 6
            checkLbl.Text = on and "✓" or ""

            local targetBG = on and C.accent or C.bgDark
            if animate then
                TweenService:Create(track, TweenInfo.new(0.15), {BackgroundColor3 = targetBG}):Play()
            else
                track.BackgroundColor3 = targetBG
            end
            knob.BackgroundTransparency = 1

        elseif style == "Fill" then
            -- Estilo fill: track inteiro muda de cor + texto ON/OFF
            trackCorner.CornerRadius = UDim.new(0, cfg.gui.cornerRadius)
            knobCorner.CornerRadius = UDim.new(0, cfg.gui.cornerRadius)
            knob.Size = UDim2.new(1, 0, 1, 0)
            knob.Position = UDim2.new(0, 0, 0, 0)
            knob.AnchorPoint = Vector2.new(0, 0)
            knob.BackgroundTransparency = 1
            checkLbl.Visible = true
            checkLbl.TextSize = H - 8
            checkLbl.Text = on and "ON" or "OFF"
            checkLbl.Font = Enum.Font.GothamBold

            local targetBG = on and C.accent or C.bgDark
            local targetTC = on and Color3.new(1,1,1) or C.muted
            if animate then
                TweenService:Create(track, TweenInfo.new(0.15), {BackgroundColor3 = targetBG}):Play()
                TweenService:Create(checkLbl, TweenInfo.new(0.15), {TextColor3 = targetTC}):Play()
            else
                track.BackgroundColor3 = targetBG
                checkLbl.TextColor3 = targetTC
            end

        elseif style == "Slash" then
            -- Estilo slash: mostra / quando OFF, — quando ON
            trackCorner.CornerRadius = UDim.new(0, cfg.gui.cornerRadius)
            knobCorner.CornerRadius = UDim.new(0, cfg.gui.cornerRadius)
            knob.Size = UDim2.new(1, -4, 1, -4)
            knob.Position = UDim2.new(0, 2, 0, 2)
            knob.AnchorPoint = Vector2.new(0, 0)
            checkLbl.Visible = true
            checkLbl.TextSize = H - 4
            checkLbl.Text = on and "—" or "/"
            checkLbl.Font = Enum.Font.GothamBold

            local targetBG = on and C.accent or C.bgDark
            knob.BackgroundTransparency = 1
            if animate then
                TweenService:Create(track, TweenInfo.new(0.15), {BackgroundColor3 = targetBG}):Play()
            else
                track.BackgroundColor3 = targetBG
            end
        end
    end

    -- Inicializar visual
    updateVisual(initState, false)

    -- Click handler
    local btn = Instance.new("TextButton")
    btn.Name = "ToggleBtn"
    btn.Size = UDim2.new(1, 0, 1, 0)
    btn.BackgroundTransparency = 1
    btn.Text = ""
    btn.ZIndex = 5
    btn.Parent = row

    btn.MouseButton1Click:Connect(function()
        local newState = not state.on
        updateVisual(newState, true)
        if callback then
            callback(newState)
        end
    end)

    -- Hover effect
    btn.MouseEnter:Connect(function()
        TweenService:Create(lbl, TweenInfo.new(0.1), {TextColor3 = C.accent}):Play()
    end)
    btn.MouseLeave:Connect(function()
        TweenService:Create(lbl, TweenInfo.new(0.1), {TextColor3 = C.text}):Play()
    end)

    -- Registar para theme updates
    table.insert(obj.themeElements, {
        type = "toggle",
        track = track,
        lbl = lbl,
        state = state,
        updateVisual = updateVisual,
    })

    return {
        frame = row,
        track = track,
        knob = knob,
        lbl = lbl,
        state = state,
        update = function(on)
            updateVisual(on, true)
        end,
        set = function(on)
            updateVisual(on, false)
        end,
    }
end

-- ── mkSlider ──────────────────────────────────────────────────
-- Estilos: "Bar" | "Thin" | "Thick" | "Glow"
-- Retorna: {frame, lbl, fill, knob, update(val)}

local function mkSlider(parent, labelText, icon, initVal, minV, maxV, x, y, callback, style)
    style = style or cfg.gui.sliderStyle
    local TRACK_H = cfg.gui.sliderH
    local ROW_H = TRACK_H + 26

    -- Track height por estilo
    local trackH = TRACK_H
    if style == "Thin" then trackH = math.max(4, TRACK_H - 6)
    elseif style == "Thick" then trackH = math.max(TRACK_H, 16)
    end

    local row = Instance.new("Frame")
    row.Name = "Slider_" .. labelText
    row.Size = UDim2.new(1, -28, 0, ROW_H)
    row.Position = UDim2.new(0, 14, 0, y)
    row.BackgroundTransparency = 1
    row.ClipsDescendants = false
    row.Parent = parent

    -- Icon + Label + Value
    local lbl = Instance.new("TextLabel")
    lbl.Name = "SliderLabel"
    lbl.Size = UDim2.new(1, -50, 0, 20)
    lbl.BackgroundTransparency = 1
    lbl.Font = Enum.Font.GothamMedium
    lbl.TextSize = cfg.gui.fontSize
    lbl.TextColor3 = C.text
    lbl.TextXAlignment = Enum.TextXAlignment.Left
    lbl.Text = (icon and icon .. "  " or "") .. labelText
    lbl.Parent = row

    local valLbl = Instance.new("TextLabel")
    valLbl.Name = "Value"
    valLbl.Size = UDim2.new(0, 46, 0, 20)
    valLbl.Position = UDim2.new(1, -46, 0, 0)
    valLbl.BackgroundColor3 = C.accent
    valLbl.BackgroundTransparency = 0.85
    valLbl.BorderSizePixel = 0
    valLbl.Font = Enum.Font.GothamBold
    valLbl.TextSize = cfg.gui.fontSize - 1
    valLbl.TextColor3 = C.accent
    valLbl.Text = tostring(initVal)
    valLbl.Parent = row
    Instance.new("UICorner", valLbl).CornerRadius = UDim.new(0, 6)

    -- Track
    local track = Instance.new("Frame")
    track.Name = "Track"
    track.Size = UDim2.new(1, 0, 0, trackH)
    track.Position = UDim2.new(0, 0, 0, 24)
    track.BackgroundColor3 = C.bgDark
    track.BorderSizePixel = 0
    track.ClipsDescendants = false
    track.Parent = row
    Instance.new("UICorner", track).CornerRadius = UDim.new(1, 0)

    -- Fill
    local pct = math.clamp((initVal - minV) / (maxV - minV), 0, 1)
    local fill = Instance.new("Frame")
    fill.Name = "Fill"
    fill.Size = UDim2.new(pct, 0, 1, 0)
    fill.BackgroundColor3 = C.accent
    fill.BorderSizePixel = 0
    fill.ClipsDescendants = false
    fill.Parent = track
    Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)

    -- Glow effect para estilo Glow
    if style == "Glow" then
        local glow = Instance.new("Frame")
        glow.Name = "Glow"
        glow.Size = UDim2.new(1, 8, 1, 8)
        glow.Position = UDim2.new(0, -4, 0, -4)
        glow.BackgroundColor3 = C.accent
        glow.BackgroundTransparency = 0.75
        glow.BorderSizePixel = 0
        glow.ZIndex = 0
        glow.Parent = fill
        Instance.new("UICorner", glow).CornerRadius = UDim.new(1, 0)

        table.insert(obj.themeElements, {type = "sliderGlow", instance = glow})
    end

    -- Knob
    local knobSize = trackH + 6
    local knob = Instance.new("Frame")
    knob.Name = "Knob"
    knob.Size = UDim2.new(0, knobSize, 0, knobSize)
    knob.AnchorPoint = Vector2.new(0.5, 0.5)
    knob.Position = UDim2.new(pct, 0, 0.5, 0)
    knob.BackgroundColor3 = Color3.new(1, 1, 1)
    knob.BorderSizePixel = 0
    knob.ZIndex = 10
    knob.ClipsDescendants = false
    knob.Parent = track
    Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)

    local knobStroke = Instance.new("UIStroke", knob)
    knobStroke.Color = C.accent
    knobStroke.Thickness = 2

    -- Drag logic com UserInputService global
    local dragging = false

    local function applyDrag(mouseX)
        local bx = track.AbsolutePosition.X
        local bw = track.AbsoluteSize.X
        if bw <= 0 then return end
        local p = math.clamp((mouseX - bx) / bw, 0, 1)
        local val = math.floor(minV + p * (maxV - minV) + 0.5)
        p = (val - minV) / (maxV - minV)

        fill.Size = UDim2.new(p, 0, 1, 0)
        knob.Position = UDim2.new(p, 0, 0.5, 0)
        valLbl.Text = tostring(val)

        if callback then
            callback(val)
        end
    end

    track.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            applyDrag(input.Position.X)
            -- Knob grow animation
            TweenService:Create(knob, TweenInfo.new(0.15, Enum.EasingStyle.Back), {
                Size = UDim2.new(0, knobSize + 4, 0, knobSize + 4)
            }):Play()
        end
    end)

    knob.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            TweenService:Create(knob, TweenInfo.new(0.15, Enum.EasingStyle.Back), {
                Size = UDim2.new(0, knobSize + 4, 0, knobSize + 4)
            }):Play()
        end
    end)

    addConnection("slider_" .. labelText .. "_move", UIS.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or
           input.UserInputType == Enum.UserInputType.Touch) then
            applyDrag(input.Position.X)
        end
    end))

    addConnection("slider_" .. labelText .. "_end", UIS.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            if dragging then
                dragging = false
                TweenService:Create(knob, TweenInfo.new(0.15), {
                    Size = UDim2.new(0, knobSize, 0, knobSize)
                }):Play()
            end
        end
    end))

    -- Registar para theme updates
    table.insert(obj.themeElements, {
        type = "slider",
        fill = fill,
        knobStroke = knobStroke,
        valLbl = valLbl,
    })

    return {
        frame = row,
        lbl = lbl,
        valLbl = valLbl,
        fill = fill,
        knob = knob,
        update = function(val)
            local p = math.clamp((val - minV) / (maxV - minV), 0, 1)
            fill.Size = UDim2.new(p, 0, 1, 0)
            knob.Position = UDim2.new(p, 0, 0.5, 0)
            valLbl.Text = tostring(val)
        end,
    }
end

local function mkPartSelector(parent, y, currentPart, callback)

    row.Position = UDim2.new(0, 14, 0, y)

    local function setActive(part)
        for nm, b in pairs(btns) do

-- Função genérica para tornar qualquer frame draggável
local function makeDraggable(triggerFrame, moveFrame, smoothTime)
    smoothTime = smoothTime or 0.06
    local dragging = false
    local dragStart = nil
    local startPos = nil

    triggerFrame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = moveFrame.Position
        end
    end)

    triggerFrame.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end)

    addConnection("drag_" .. triggerFrame.Name, UIS.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or
           input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - dragStart
            local targetPos = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
            -- Interpolação suave
            TweenService:Create(moveFrame, TweenInfo.new(smoothTime), {
                Position = targetPos
            }):Play()
        end
    end))
end

-- ── Tab Switch Function ────────────────────────────────────
local function switchTab(tabId)
    if tabId == currentTab then return end

    -- Esconder tab atual
    if tabFrames[currentTab] then
        tabFrames[currentTab].Visible = false
    end
    -- Descolorir botão antigo
    if tabButtons[currentTab] then
        TweenService:Create(tabButtons[currentTab], TweenInfo.new(0.15), {
            BackgroundTransparency = 1,
            TextColor3 = C.muted,
        }):Play()
    end

    currentTab = tabId

    -- Mostrar nova tab
    if tabFrames[tabId] then
        tabFrames[tabId].Visible = true

-- ── FPS Counter ─────────────────────────────────────────────
local frameCount = 0
local lastFPSUpdate = tick()

addConnection("fps_counter", RunService.RenderStepped:Connect(function()
    frameCount = frameCount + 1
    local now = tick()
    if now - lastFPSUpdate >= 1 then
        obj.currentFPS = math.floor(frameCount / (now - lastFPSUpdate))
        frameCount = 0
        lastFPSUpdate = now
        pcall(function()
            fpsLbl.Text = obj.currentFPS .. " FPS"
        end)
    end
end))

    while st.running do
        pcall(function()
            local stats = game:GetService("Stats")
            local ping = math.floor(stats:GetValue("DataReceive") or 0)
            if ping <= 0 then
                local nStats = stats:FindFirstChild("PerformanceStats")
                if nStats then
                    local p = nStats:FindFirstChild("Ping")
                    if p then ping = math.floor(p:GetValue()) end
                end
            end
            obj.currentPing = ping
            pingLbl.Text = ping .. "ms"
        end)
        task.wait(2)
    end
end)

-- ── Validação de alvo ──────────────────────────────────────────
local function isValidTarget(plr)
    if not plr or plr == player or not plr.Character then return false end
    if not plr.Character.Parent then return false end
    local hum = plr.Character:FindFirstChildOfClass("Humanoid")
    if not hum or hum.Health <= 0 then return false end
    -- Team check
    if st.teamCheck and plr.Team and player.Team and plr.Team == player.Team then return false end
    -- Health check
    if st.healthCheck and hum.MaxHealth > 0 then
        local pct = (hum.Health / hum.MaxHealth) * 100
        if pct < cfg.healthMin then return false end
    end
    -- Distance check
    local myChar = player.Character
    if not myChar then return false end
    local myHRP = myChar:FindFirstChild("HumanoidRootPart")
    local theirHRP = plr.Character:FindFirstChild("HumanoidRootPart")
    if myHRP and theirHRP then
        local dist = (myHRP.Position - theirHRP.Position).Magnitude
        if dist > cfg.maxDistance then return false end
    end
    return true
end

-- ── Obter parte alvo ──────────────────────────────────────────
local function getAimPart(char)
    if not char then return nil end
    local part = cfg.hitPart
    if part == "Random" then
        local parts = {"Head", "UpperTorso", "HumanoidRootPart"}
        part = parts[math.random(1, #parts)]
    end
    if part == "Head" then
        return char:FindFirstChild("Head")
    elseif part == "Torso" then
        return char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso") or char:FindFirstChild("HumanoidRootPart")
    end
    return char:FindFirstChild("Head") or char:FindFirstChild("HumanoidRootPart")
end

-- ── Visible check (Raycast) ───────────────────────────────────
local function isVisible(targetPart)
    if not st.visibleCheck then return true end
    if not targetPart then return false end
    local myChar = player.Character
    if not myChar then return false end
    local origin = camera.CFrame.Position
    local direction = (targetPart.Position - origin)
    local rp = RaycastParams.new()
    rp.FilterType = Enum.RaycastFilterType.Exclude
    rp.FilterDescendantsInstances = {myChar, camera}
    local result = Workspace:Raycast(origin, direction, rp)
    if not result then return true end
    return result.Instance:IsDescendantOf(targetPart.Parent)
end

-- ── Prediction (velocity-based) ───────────────────────────────
local function predictPosition(targetPart)
    if not st.prediction or not targetPart then
        return targetPart and targetPart.Position or Vector3.zero
    end
    local vel = Vector3.zero
    pcall(function()
        if targetPart:IsA("BasePart") then
            vel = targetPart.AssemblyLinearVelocity or targetPart.Velocity or Vector3.zero
        end
    end)
    -- Predict based on distance (further = more prediction)
    local dist = (targetPart.Position - camera.CFrame.Position).Magnitude
    local predTime = math.clamp(dist / 800, 0.01, 0.15) * (cfg.predictionAmount / 50)
    return targetPart.Position + vel * predTime
end

-- ── Encontrar melhor alvo no FOV ──────────────────────────────
local rmbDown = false
local function closestInFOV()
    local mp = UIS:GetMouseLocation()
    local best, bestD = nil, math.huge
    for _, plr in ipairs(Players:GetPlayers()) do
        if isValidTarget(plr) then
            local part = getAimPart(plr.Character)
            if part and isVisible(part) then
                local pos = predictPosition(part)
                local sp, onScreen = camera:WorldToViewportPoint(pos)
                if onScreen then
                    local d = (Vector2.new(sp.X, sp.Y) - mp).Magnitude
                    if d < cfg.aimbotFOV and d < bestD then
                        best = plr; bestD = d
                    end
                end
            end
        end
    end
    return best
end

-- ── Aimbot RenderStepped ──────────────────────────────────────
local aimConn = addConnection(RS.RenderStepped:Connect(function()
    if not st.running then return end
    
    if st.aimbot and rmbDown then
        -- Acquire / maintain target
        if obj.lockedTarget then
            if not isValidTarget(obj.lockedTarget) then
                obj.lockedTarget = nil
            end
        end
        if not obj.lockedTarget then
            obj.lockedTarget = closestInFOV()
        end
    else
        obj.lockedTarget = nil
    end
    
    -- Pull camera
    if st.aimbot and rmbDown and obj.lockedTarget and obj.lockedTarget.Character then
        local part = getAimPart(obj.lockedTarget.Character)
        if part then
            local targetPos = predictPosition(part)
            if cfg.aimSmooth == 0 then
                camera.CFrame = CFrame.new(camera.CFrame.Position, targetPos)
            else
                local t = (1 - cfg.aimSmooth / 100) * 0.93 + 0.02
                local curDir = camera.CFrame.LookVector
                local tgtDir = (targetPos - camera.CFrame.Position).Unit
                local blended = curDir:Lerp(tgtDir, t).Unit
                camera.CFrame = CFrame.new(camera.CFrame.Position, camera.CFrame.Position + blended)
            end
        else
            obj.lockedTarget = nil
        end
    end
end))

        end

        local p0 = char:FindFirstChild(pair[1])
        local p1 = char:FindFirstChild(pair[2])
        if p0 and p1 and p0:IsA("BasePart") and p1:IsA("BasePart") then
            local a0 = Instance.new("Attachment")
            a0.Name = "MSkel0_" .. i
            a0.Parent = p0
            local a1 = Instance.new("Attachment")
            a1.Name = "MSkel1_" .. i
            a1.Parent = p1
            local beam = Instance.new("Beam")
            beam.Name = "MSkelBeam_" .. i
            beam.Attachment0 = a0
            beam.Attachment1 = a1
            beam.Width0 = 0.15
            beam.Width1 = 0.15
            beam.Color = ColorSequence.new(C.accent)
            beam.FaceCamera = true
            beam.Transparency = NumberSequence.new(0.2)
            beam.Parent = p0
            
            d.skelParts[i] = {beam = beam, a0 = a0, a1 = a1}
        end
    end
end

local function removeSkeleton(plr)

    if d and d.skelParts then
        for _, s in pairs(d.skelParts) do

            pcall(function() s.a0:Destroy() end)
            pcall(function() s.a1:Destroy() end)
        end
        d.skelParts = nil
    end
end

local function enableFly()
    local char = player.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    -- Remove existing
    for _, c in pairs(hrp:GetChildren()) do
        if c:IsA("BodyVelocity") and c.Name == "MFlyVel" then c:Destroy() end
        if c:IsA("BodyGyro") and c.Name == "MFlyGyro" then c:Destroy() end
    end
    obj.bv = Instance.new("BodyVelocity")
    obj.bv.Name = "MFlyVel"
    obj.bv.MaxForce = Vector3.new(1e5, 1e5, 1e5)
    obj.bv.Velocity = Vector3.zero

    obj.bg = Instance.new("BodyGyro")

    obj.bg.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
    obj.bg.P = 1e4
    obj.bg.Parent = hrp
end

local function disableFly()
    if obj.bv then pcall(function() obj.bv:Destroy() end); obj.bv = nil end

local HITBOX_PARTS = {
    "Head", "UpperTorso", "LowerTorso", "HumanoidRootPart",
    "LeftUpperArm", "LeftLowerArm", "LeftHand",
    "RightUpperArm", "RightLowerArm", "RightHand",
    "LeftUpperLeg", "LeftLowerLeg", "LeftFoot",
    "RightUpperLeg", "RightLowerLeg", "RightFoot",
}

local function resetHitbox(plr)
    local s = obj.origSizes[plr]
    if not s or not plr.Character then return end
    for nm, sz in pairs(s) do
        local p = plr.Character:FindFirstChild(nm)
        if p and p:IsA("BasePart") then
            pcall(function()
                p.Size = sz
                p.Transparency = (nm == "HumanoidRootPart") and 1 or 0
                p.CanCollide = false
            end)
        end
    end
    obj.origSizes[plr] = nil
end

local function resetAllHitboxes()
    for p in pairs(obj.origSizes) do resetHitbox(p) end
end

local hitLoop = task.spawn(function()
    while st.running do
        task.wait(0.8)
        if st.hitbox then
            for _, plr in ipairs(Players:GetPlayers()) do
                if plr ~= player and plr.Character and isValidTarget(plr) then
                    obj.origSizes[plr] = obj.origSizes[plr] or {}
                    for _, nm in ipairs(HITBOX_PARTS) do
                        local p = plr.Character:FindFirstChild(nm)
                        if p and p:IsA("BasePart") then
                            pcall(function()
                                if not obj.origSizes[plr][nm] then
                                    obj.origSizes[plr][nm] = p.Size
                                end
                                p.Size = obj.origSizes[plr][nm] * (cfg.hitboxSize / 5)
                                p.Transparency = cfg.hitboxTransp
                                p.CanCollide = false
                            end)
                        end
                    end
                elseif plr ~= player then
                    resetHitbox(plr)
                end
            end
        else
            resetAllHitboxes()
        end
    end
end)

addConnection(Players.PlayerRemoving:Connect(function(plr) resetHitbox(plr) end))

                if child:IsA("Frame") and child.Name ~= "Shadow" then

-- ── Main Input Handler ────────────────────────────────────────
addConnection(UIS.InputBegan:Connect(function(input, gp)
    if gp or not st.running then return end
    
    -- RMB for aimbot
    if input.UserInputType == Enum.UserInputType.MouseButton2 then
        if st.aimbot then rmbDown = true end
        return
    end
    
    if not input.KeyCode then return end
    local kc = input.KeyCode
    
    -- Check each keybind
    for action, bind in pairs(keybinds) do
        if kc == bind and toggleActions[action] then
            local newState = not st[action]
            toggleActions[action](newState)
            -- Update UI toggles
            if obj.toggleFrames and obj.toggleFrames[action] then
                pcall(function()
                    local tFrame = obj.toggleFrames[action]
                    local knob = tFrame:FindFirstChild("Knob")
                    local bg = tFrame
                    if knob then
                        local xPos = newState and UDim2.new(1, -cfg.gui.toggleH + 4, 0.5, 0) or UDim2.new(0, 4, 0.5, 0)
                        TS:Create(knob, TweenInfo.new(0.2, Enum.EasingStyle.Back), {Position = xPos}):Play()
                    end
                    bg.BackgroundColor3 = newState and C.accent or C.bgDark
                end)
            end
            return
        end
    end
    
    -- Panic key (End)
    if kc == Enum.KeyCode.End then
        -- Disable everything
        for action, fn in pairs(toggleActions) do
            if st[action] and action ~= "gui" then
                fn(false)
            end
        end
        spDisable()
        notify("🚨 PANIC — Tudo desativado!", C.error)
        return
    end
    
    -- Eject (P)
    if kc == keybinds.eject then
        notify("⚠️ Ejecting...", C.error)
        task.delay(0.5, function()
            -- Disable all features
            st.running = false
            clearESP()
            resetAllHitboxes()
            disableFly()
            disableFullbright()
            spDisable()
            
            -- Reset character
            pcall(function()
                local hum = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
                if hum then hum.WalkSpeed = 16 end
            end)
            pcall(function()
                local char = player.Character
                if char then
                    for _, p in ipairs(char:GetDescendants()) do
                        if p:IsA("BasePart") then p.CanCollide = true end
                    end
                end
            end)
            
            -- Destroy all connections
            for _, cn in ipairs(obj.connections) do
                pcall(function() cn:Disconnect() end)
            end
            
            -- Cleanup fly objects
            pcall(function()
                local hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
                if hrp then
                    for _, o in pairs(hrp:GetChildren()) do
                        if o:IsA("BodyVelocity") or o:IsA("BodyGyro") then o:Destroy() end
                    end
                end
            end)
            
            -- Reset camera
            pcall(function()
                local hum = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
                if hum then camera.CameraSubject = hum end
            end)
            
            -- Destroy GUIs
            for _, gui in ipairs(SP.allGuis) do
                pcall(function() gui:Destroy() end)
            end
            pcall(function()
                for _, g in pairs(getService("CoreGui"):GetChildren()) do
                    if g.Name and g.Name:match("^MESP_") then g:Destroy() end
                    if g.Name and g.Name:match("^Medusa") then g:Destroy() end
                end
            end)
            
            -- Clean env
            pcall(function() getgenv().MedusaLoaded = false end)
            
            print("[MEDUSA] ✅ Ejected cleanly!")
        end)
        return
    end
end))

local chParts = {} -- crosshair lines/dots

local function updateCrosshair()
    -- Clear old
    for _, p in ipairs(chParts) do pcall(function() p:Destroy() end) end
    chParts = {}
    
    if not st.crosshair then return end
    
    local vp = camera.ViewportSize
    local cx, cy = vp.X / 2, vp.Y / 2
    local size = cfg.crosshairSize
    local gap = cfg.crosshairGap
    local style = cfg.crosshairStyle
    local color = C.accent
    
    local function mkLine(x, y, w, h)
        local l = Instance.new("Frame")
        l.Size = UDim2.new(0, w, 0, h)
        l.Position = UDim2.new(0, x, 0, y)
        l.BackgroundColor3 = color
        l.BorderSizePixel = 0
        l.Parent = chFrame
        table.insert(chParts, l)
        return l
    end
    
    if style == "Cross" then
        mkLine(cx - size - gap, cy - 1, size, 2) -- left
        mkLine(cx + gap, cy - 1, size, 2)         -- right
        mkLine(cx - 1, cy - size - gap, 2, size) -- top
        mkLine(cx - 1, cy + gap, 2, size)         -- bottom
    elseif style == "Dot" then
        local dot = mkLine(cx - 3, cy - 3, 6, 6)
        Instance.new("UICorner", dot).CornerRadius = UDim.new(1, 0)
    elseif style == "Circle" then
        local circle = Instance.new("Frame")
        circle.Size = UDim2.new(0, size * 2, 0, size * 2)
        circle.Position = UDim2.new(0, cx - size, 0, cy - size)
        circle.BackgroundTransparency = 1
        circle.BorderSizePixel = 0
        circle.Parent = chFrame
        Instance.new("UICorner", circle).CornerRadius = UDim.new(1, 0)
        local sk = Instance.new("UIStroke", circle)
        sk.Color = color; sk.Thickness = 2
        table.insert(chParts, circle)
        -- Center dot
        local dot = mkLine(cx - 2, cy - 2, 4, 4)
        Instance.new("UICorner", dot).CornerRadius = UDim.new(1, 0)
    elseif style == "T-Cross" then
        mkLine(cx - size - gap, cy - 1, size, 2)  -- left
        mkLine(cx + gap, cy - 1, size, 2)          -- right
        mkLine(cx - 1, cy - size - gap, 2, size)  -- top only
    end
end

    -- Only update if crosshair state changed (avoid spam)
end))

local function spectatePlayer(plr)
    if not plr or not plr.Character then return end
    local hum = plr.Character:FindFirstChildOfClass("Humanoid")
    if hum then
        camera.CameraSubject = hum
        obj.spectating = plr

local function unspectate()
    pcall(function()
        local hum = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
        if hum then camera.CameraSubject = hum end
    end)
    if obj.spectating then
        notify("📷 Unspectated", C.textMuted)
    end
    obj.spectating = nil
end

local function flingPlayer(plr)
    if not plr or not plr.Character then return end
    local myChar = player.Character
    local myHRP = myChar and myChar:FindFirstChild("HumanoidRootPart")
    local theirHRP = plr.Character:FindFirstChild("HumanoidRootPart")
    if not myHRP or not theirHRP then return end
    
    local origPos = myHRP.CFrame
    -- Teleport to target
    myHRP.CFrame = theirHRP.CFrame * CFrame.new(0, 0, -2)
    -- Apply velocity
    pcall(function()
        myHRP.Velocity = Vector3.new(math.random(-200, 200), 300, math.random(-200, 200))
        myHRP.RotVelocity = Vector3.new(math.random(-50, 50), math.random(-50, 50), math.random(-50, 50))
    end)
    -- Return after delay
    task.delay(0.3, function()
        pcall(function() myHRP.CFrame = origPos end)
    end)
    notify("💥 Flung " .. plr.DisplayName, C.warning)
end

local function tpToPlayer(plr)
    if not plr or not plr.Character then return end
    local myHRP = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    local theirHRP = plr.Character:FindFirstChild("HumanoidRootPart")
    if myHRP and theirHRP then
        myHRP.CFrame = theirHRP.CFrame * CFrame.new(0, 0, -3)
        notify("🏃 TP to " .. plr.DisplayName, C.success)
    end
end

local function serverHop()
    notify("🌐 A procurar servidor...", C.accent)
    task.spawn(function()
        pcall(function()
            local TS = game:GetService("TeleportService")
            local Http = game:GetService("HttpService")
            local url = "https://games.roblox.com/v1/games/" .. game.GameId .. "/servers/Public?sortOrder=Asc&limit=100"
            local body = game:HttpGet(url)
            local data = Http:JSONDecode(body)
            if data and data.data then
                for _, sv in ipairs(data.data) do
                    if sv.playing and sv.maxPlayers and sv.playing < sv.maxPlayers and sv.id ~= game.JobId then
                        TS:TeleportToPlaceInstance(game.PlaceId, sv.id, player)
                        return
                    end
                end
            end
            notify("❌ Nenhum servidor encontrado", C.error)
        end)
    end)
end

    local link = "https://www.roblox.com/games/" .. game.PlaceId
    pcall(function()
        if setclipboard then setclipboard(link)
        elseif toclipboard then toclipboard(link)
        else return end
        notify("📋 Link copiado!", C.success)
    end)
end
