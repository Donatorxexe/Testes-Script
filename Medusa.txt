local function applyTheme(name)
    local t = themes[name]
    if not t then return end
    C.accent = t.accent
    C.bg = t.bg
    C.bgCard = Color3.fromRGB(
        math.min(255, math.floor(t.bg.R*255)+8),
        math.min(255, math.floor(t.bg.G*255)+8),
        math.min(255, math.floor(t.bg.B*255)+10)
    )
    C.card = C.bgCard
    C.bgDark = Color3.fromRGB(
        math.max(0, math.floor(t.bg.R*255)-4),
        math.max(0, math.floor(t.bg.G*255)-4),
        math.max(0, math.floor(t.bg.B*255)-4)
    )
    for _, el in ipairs(obj.themeElements) do
        pcall(function()
            if el.typ == "bg" then el.obj.BackgroundColor3 = C.bg
            elseif el.typ == "card" then el.obj.BackgroundColor3 = C.bgCard
            elseif el.typ == "dark" then el.obj.BackgroundColor3 = C.bgDark
            elseif el.typ == "accent" then el.obj.BackgroundColor3 = C.accent
            elseif el.typ == "accentText" then el.obj.TextColor3 = C.accent
            elseif el.typ == "stroke" then el.obj.Color = C.accent
            elseif el.typ == "border" then el.obj.Color = C.border
            end
        end)
    end
end

-- Save/Load config
local function saveConfig()
    if not EXEC.hasWriteFile then return end
    pcall(function()
        local data = game:GetService("HttpService"):JSONEncode({
            theme = {r=math.floor(C.accent.R*255), g=math.floor(C.accent.G*255), b=math.floor(C.accent.B*255)},
            gui = cfg.gui,
        })
        writefile("MedusaConfig.json", data)
    end)
end

local function loadConfig()
    if not EXEC.hasReadFile or not EXEC.hasIsFile then return end
    pcall(function()
        if isfile("MedusaConfig.json") then
            local data = game:GetService("HttpService"):JSONDecode(readfile("MedusaConfig.json"))
            if data.theme then C.accent = Color3.fromRGB(data.theme.r, data.theme.g, data.theme.b) end
            if data.gui then for k,v in pairs(data.gui) do cfg.gui[k] = v end end
        end
    end)
end
loadConfig()

local SP = {
    active = false,
    method = "none",
    hiddenParent = nil,
    allGuis = {},
}

-- Detect best method
do
    local function tryGethui()
        local fns = {
            typeof(gethui) == "function" and gethui,
            getgenv and typeof(getgenv().gethui) == "function" and getgenv().gethui,
            typeof(get_hidden_gui) == "function" and get_hidden_gui,
            typeof(gethiddenui) == "function" and gethiddenui,
        }
        for _, fn in ipairs(fns) do
            if fn then
                local ok, result = pcall(fn)
                if ok and result then
                    SP.method = "gethui"
                    SP.hiddenParent = result
                    print("[Medusa SP] ✅ Method: gethui()")
                    return true
                end
            end
        end
        return false
    end

    local function tryProtectCoreGui()
        if CoreGui then
            local protectFn = typeof(protect_gui) == "function" and protect_gui
                or (typeof(syn) == "table" and typeof(syn.protect_gui) == "function" and syn.protect_gui)
            if protectFn then
                SP.method = "protect+CoreGui"
                SP.hiddenParent = CoreGui
                SP._protectFn = protectFn
                print("[Medusa SP] ✅ Method: protect_gui + CoreGui")
                return true
            end
        end
        return false
    end

    local function tryClonerefCoreGui()
        if EXEC.hasCloneref and CoreGui then
            local ok, ref = pcall(cloneref, CoreGui)
            if ok and ref then
                SP.method = "cloneref"
                SP.hiddenParent = ref
                print("[Medusa SP] ✅ Method: cloneref(CoreGui)")
                return true
            end
        end
        return false
    end

    if not tryGethui() then
        if not tryProtectCoreGui() then
            if not tryClonerefCoreGui() then
                print("[Medusa SP] ⚠️ No StreamProof method available")
            end
        end
    end
end

local function createGui(name)
    local sg = Instance.new("ScreenGui")
    sg.Name = name or "MedusaGui"
    sg.ResetOnSpawn = false
    sg.DisplayOrder = 999999
    sg.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    -- Discord/OBS protection
    pcall(function() sg.IgnoreGuiInset = true end)
    -- Try to use CaptureService exclusion (modern Roblox)
    pcall(function()
        local cs = game:GetService("CaptureService")
        if cs and cs.ExcludeFromCapture then
            cs:ExcludeFromCapture(sg)
        end
    end)
    if SP.active and SP.hiddenParent then
        if SP._protectFn then pcall(SP._protectFn, sg) end
        sg.Parent = SP.hiddenParent
    else
        sg.Parent = playerGui
    end
    table.insert(SP.allGuis, sg)
    return sg
end

local function spEnable()
    if SP.method == "none" then return false end
    SP.active = true
    for _, gui in ipairs(SP.allGuis) do
        pcall(function()
            if SP._protectFn then pcall(SP._protectFn, gui) end
            gui.Parent = SP.hiddenParent
        end)
    end
    print("[Medusa SP] ✅ Enabled - method:", SP.method)
    return true
end

local function spDisable()
    SP.active = false
    for _, gui in ipairs(SP.allGuis) do
        pcall(function() gui.Parent = playerGui end)
    end
    print("[Medusa SP] Disabled")
end

    ksk.Color = C.accent
    ksk.Thickness = 2

    -- Glow style
    if cfg.sliderStyle == 4 then
        fill.BackgroundTransparency = 0.3
    end

    local dragging = false
    local function applyPos(mx)
        local bx = track.AbsolutePosition.X
        local bw = track.AbsoluteSize.X
        if bw <= 0 then return end
        local p = math.clamp((mx - bx) / bw, 0, 1)
        local val = math.floor(minV + p * (maxV - minV))
        fill.Size = UDim2.new(p, 0, 1, 0)
        knob.Position = UDim2.new(p, 0, 0.5, 0)
        valLbl.Text = tostring(val)
        if callback then callback(val) end
    end

    track.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then dragging = true; applyPos(i.Position.X) end end)
    knob.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then dragging = true end end)
    addConn(UIS.InputChanged:Connect(function(i) if dragging and i.UserInputType == Enum.UserInputType.MouseMovement then applyPos(i.Position.X) end end))
    addConn(UIS.InputEnded:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end end))

    return row, lbl, valLbl
end

-- ═══ S9: DRAG SYSTEM ═══
local function makeDraggable(topElement, moveElement)
    local dragging, dragStart, startPos = false, nil, nil
    topElement.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = i.Position
            startPos = moveElement.Position
        end
    end)
    topElement.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end
    end)
    addConn(UIS.InputChanged:Connect(function(i)
        if dragging and i.UserInputType == Enum.UserInputType.MouseMovement then
            local d = i.Position - dragStart
            local target = UDim2.new(startPos.X.Scale, startPos.X.Offset + d.X, startPos.Y.Scale, startPos.Y.Offset + d.Y)
            TS:Create(moveElement, TweenInfo.new(0.06), {Position = target}):Play()
        end
    end))
end

-- ═══ S21: AIMBOT ENGINE ═══
local function isValidTarget(plr)
    if not plr or plr == player or not plr.Character then return false end
    if cfg.teamCheck and plr.Team and player.Team and plr.Team == player.Team then return false end
    local hum = plr.Character:FindFirstChildOfClass("Humanoid")
    if not hum or hum.Health <= 0 then return false end
    if cfg.healthCheck and hum.MaxHealth > 0 then
        local pct = (hum.Health / hum.MaxHealth) * 100
        if pct < cfg.healthMin then return false end
    end
    return true
end

local function getAimPart(char)
    if not char then return nil end
    if cfg.aimbotPart == "Head" then return char:FindFirstChild("Head")
    elseif cfg.aimbotPart == "Torso" then return char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso")
    elseif cfg.aimbotPart == "Random" then
        local parts = {}
        for _, nm in ipairs({"Head", "UpperTorso", "Torso", "HumanoidRootPart"}) do
            local p = char:FindFirstChild(nm)
            if p then table.insert(parts, p) end
        end
        return #parts > 0 and parts[math.random(1, #parts)] or nil
    end
    return char:FindFirstChild("Head")
end

local function predictPosition(part, char)
    if not cfg.prediction or not part then return part and part.Position or Vector3.zero end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if hrp then
        local vel = hrp.AssemblyLinearVelocity or hrp.Velocity
        if vel then
            local dist = (part.Position - camera.CFrame.Position).Magnitude
            local bulletTime = dist / 1000
            return part.Position + vel * bulletTime
        end
    end
    return part.Position
end

local function visibleCheck(part)
    if not cfg.visibleCheck or not part then return true end
    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    params.FilterDescendantsInstances = {player.Character, camera}
    local origin = camera.CFrame.Position
    local dir = (part.Position - origin)
    local result = Workspace:Raycast(origin, dir, params)
    if not result then return true end
    return result.Instance:IsDescendantOf(part.Parent)
end

local function closestInFOV()
    local mp = UIS:GetMouseLocation()
    local best, bestD = nil, math.huge
    for _, plr in ipairs(Players:GetPlayers()) do
        if isValidTarget(plr) then
            local part = getAimPart(plr.Character)
            if part then
                local dist = (part.Position - camera.CFrame.Position).Magnitude
                if dist <= cfg.maxDistance then
                    local sp, onScreen = camera:WorldToViewportPoint(part.Position)
                    if onScreen then
                        local d = (Vector2.new(sp.X, sp.Y) - mp).Magnitude
                        if d < cfg.aimbotFOV and d < bestD then
                            if visibleCheck(part) then
                                best = plr
                                bestD = d
                            end
                        end
                    end
                end
            end
        end
    end
    return best
end

-- Aimbot render loop
addConn(RS.RenderStepped:Connect(function()
    if not st.running then return end

    -- Ghost mode
    if ghostActive and obj.panel then
        local mp = UIS:GetMouseLocation()
        local pp = obj.panel.AbsolutePosition
        local ps = obj.panel.AbsoluteSize
        local inside = mp.X >= pp.X and mp.X <= pp.X + ps.X and mp.Y >= pp.Y and mp.Y <= pp.Y + ps.Y
        local target = inside and cfg.gui.panelOpacity or 0.92
        obj.panel.BackgroundTransparency = obj.panel.BackgroundTransparency + (target - obj.panel.BackgroundTransparency) * 0.1
    end

    -- Aimbot logic
    if st.aimbot and rmbDown then
        if obj.lockedTarget then
            if not isValidTarget(obj.lockedTarget) then obj.lockedTarget = nil end
        end
        if not obj.lockedTarget then obj.lockedTarget = closestInFOV() end
    else
        obj.lockedTarget = nil
    end

    if st.aimbot and rmbDown and obj.lockedTarget and obj.lockedTarget.Character then
        local part = getAimPart(obj.lockedTarget.Character)
        if part then
            local targetPos = predictPosition(part, obj.lockedTarget.Character)
            if cfg.aimSmooth == 0 then
                camera.CFrame = CFrame.new(camera.CFrame.Position, targetPos)
            else
                local t = (1 - cfg.aimSmooth / 100) * 0.93 + 0.02
                local cur = camera.CFrame.LookVector
                local tar = (targetPos - camera.CFrame.Position).Unit
                local blend = cur:Lerp(tar, t).Unit
                camera.CFrame = CFrame.new(camera.CFrame.Position, camera.CFrame.Position + blend)
            end
        end
    end

    -- Status pills update
    if obj.statusPills.esp then
        obj.statusPills.esp.Text = "👁️ ESP " .. (st.esp and "ON" or "OFF")
        obj.statusPills.esp.TextColor3 = st.esp and C.success or C.textMuted
    end
    if obj.statusPills.aimbot then
        obj.statusPills.aimbot.Text = "🎯 Aim " .. (st.aimbot and "ON" or "OFF") .. (obj.lockedTarget and " [LOCK]" or "")
        obj.statusPills.aimbot.TextColor3 = st.aimbot and C.purple or C.textMuted
    end
    if obj.statusPills.fly then
        obj.statusPills.fly.Text = "✈️ Fly " .. (st.fly and "ON" or "OFF")

-- ═══ S24: ESP ═══
local function removeESP(plr)
    local d = obj.espObjs[plr]
    if not d then return end
    pcall(function() if d.hl then d.hl:Destroy() end end)
    pcall(function() if d.bb then d.bb:Destroy() end end)
    pcall(function() if d.cn then d.cn:Disconnect() end end)
    pcall(function() if d.box then d.box:Destroy() end end)
    pcall(function() if d.tracer then d.tracer:Destroy() end end)
    pcall(function()
        if d.bones then
            for _, b in ipairs(d.bones) do pcall(function() b:Destroy() end) end
        end
    end)
    obj.espObjs[plr] = nil
end

local function clearESP()
    for p in pairs(obj.espObjs) do removeESP(p) end
end

local function addESP(plr)
    if not st.esp or not plr or plr == player or obj.espObjs[plr] then return end
    if not isValidTarget(plr) then return end
    local char = plr.Character
    if not char then return end
    local head = char:FindFirstChild("Head")
    if not head then return end

    local data = {}

    -- Highlight
    pcall(function()
        local hl = Instance.new("Highlight")
        hl.FillTransparency = 0.5
        hl.OutlineTransparency = 0
        hl.FillColor = C.accent
        hl.OutlineColor = C.accent
        hl.Adornee = char
        hl.Parent = char
        data.hl = hl
    end)

    -- Billboard
    local bb = Instance.new("BillboardGui")
    bb.Adornee = head
    bb.AlwaysOnTop = true
    bb.Size = UDim2.new(0, 140, 0, 40)
    bb.StudsOffset = Vector3.new(0, 3, 0)
    bb.Parent = CoreGui or playerGui
    data.bb = bb

    local nl = Instance.new("TextLabel")
    nl.Size = UDim2.new(1, 0, 0.5, 0)
    nl.BackgroundTransparency = 1
    nl.Font = Enum.Font.GothamBold
    nl.TextSize = 14
    nl.TextColor3 = Color3.new(1, 1, 1)
    nl.TextStrokeTransparency = 0.3
    nl.Text = plr.DisplayName
    nl.Parent = bb

    local dl = Instance.new("TextLabel")
    dl.Size = UDim2.new(1, 0, 0.5, 0)
    dl.Position = UDim2.new(0, 0, 0.5, 0)
    dl.BackgroundTransparency = 1
    dl.Font = Enum.Font.Gotham
    dl.TextSize = 12
    dl.TextColor3 = C.accent
    dl.TextStrokeTransparency = 0.3
    dl.Text = "0m"
    dl.Parent = bb

    data.cn = RS.RenderStepped:Connect(function()
        pcall(function()
            if char and char.Parent and head and head.Parent then
                local dist = math.floor((head.Position - camera.CFrame.Position).Magnitude)
                if dist > cfg.espDistance then
                    bb.Enabled = false
                    if data.hl then data.hl.Enabled = false end
                else
                    bb.Enabled = true
                    if data.hl then data.hl.Enabled = true end
                    dl.Text = dist .. "m"
                end
            else
                removeESP(plr)
            end
        end)
    end)

    -- 3D Box
    if st.box3d then
        pcall(function()
            local hrp = char:FindFirstChild("HumanoidRootPart")
            if hrp then
                local sel = Instance.new("SelectionBox")
                sel.Adornee = hrp
                sel.Color3 = C.accent
                sel.LineThickness = 0.03
                sel.Transparency = 0.3
                sel.Parent = hrp
                data.box = sel
            end
        end)
    end

    obj.espObjs[plr] = data
end

local function enableFly()

    local hrp = char:FindFirstChild("HumanoidRootPart")

    obj.bv = Instance.new("BodyVelocity")

end

        local char = player.Character

        obj.bv.Velocity = mv.Magnitude > 0 and mv.Unit * cfg.flySpeed or Vector3.zero
        obj.bg.CFrame = cam

local function resetHitbox(plr)
    local s = obj.origSizes[plr]
    if not s or not plr.Character then return end
    for nm, sz in pairs(s) do
        local p = plr.Character:FindFirstChild(nm)

            pcall(function() p.Size = sz; p.Transparency = (nm == "HumanoidRootPart") and 1 or 0 end)
        end

    obj.origSizes[plr] = nil
end

    for p in pairs(obj.origSizes) do resetHitbox(p) end
end

local function toggleFeature(key)
    if key == "esp" then
        st.esp = not st.esp
        if not st.esp then clearESP() else lastESPRefresh = os.time() end
        notify(st.esp and "👁️ ESP ON" or "❌ ESP OFF", st.esp and C.success or C.error)
    elseif key == "aimbot" then
        st.aimbot = not st.aimbot
        if not st.aimbot then rmbDown = false; obj.lockedTarget = nil end
        notify(st.aimbot and "🎯 Aimbot ON" or "❌ Aimbot OFF", st.aimbot and C.purple or C.error)
    elseif key == "silentAim" then
        st.silentAim = not st.silentAim
        notify(st.silentAim and "🔇 Silent Aim ON" or "❌ Silent Aim OFF", st.silentAim and C.purple or C.error)
    elseif key == "triggerBot" then
        st.triggerBot = not st.triggerBot
        notify(st.triggerBot and "🔫 Trigger Bot ON" or "❌ Trigger Bot OFF", st.triggerBot and C.warning or C.error)
    elseif key == "fly" then
        st.fly = not st.fly
        if st.fly then enableFly() else disableFly() end
        notify(st.fly and "✈️ Fly ON" or "❌ Fly OFF", st.fly and C.blue or C.error)
    elseif key == "noclip" then
        st.noclip = not st.noclip
        notify(st.noclip and "👻 Noclip ON" or "❌ Noclip OFF", st.noclip and C.success or C.error)
    elseif key == "hitbox" then
        st.hitbox = not st.hitbox
        if not st.hitbox then resetAllHitboxes() end
        notify(st.hitbox and "📦 Hitbox ON" or "❌ Hitbox OFF", st.hitbox and C.warning or C.error)
    elseif key == "speed" then
        st.speed = not st.speed
        notify(st.speed and "🏃 Speed ON" or "❌ Speed OFF", st.speed and C.success or C.error)
    elseif key == "infJump" then
        st.infJump = not st.infJump
        notify(st.infJump and "🦘 Inf Jump ON" or "❌ Inf Jump OFF", st.infJump and C.success or C.error)
    elseif key == "fullbright" then
        st.fullbright = not st.fullbright
        setFullbright(st.fullbright)
        notify(st.fullbright and "💡 Fullbright ON" or "❌ Fullbright OFF", st.fullbright and C.success or C.error)
    elseif key == "noFallDmg" then
        st.noFallDmg = not st.noFallDmg
        notify(st.noFallDmg and "🪂 No Fall Dmg ON" or "❌ No Fall Dmg OFF", st.noFallDmg and C.success or C.error)
    elseif key == "clickTP" then
        st.clickTP = not st.clickTP
        notify(st.clickTP and "🖱️ Click TP ON" or "❌ Click TP OFF", st.clickTP and C.success or C.error)
    elseif key == "streamProof" then

            if not spEnable() then st.streamProof = false; notify("⚠️ StreamProof unavailable", C.error) end
        else spDisable() end

    elseif key == "ghostMode" then
        st.ghostMode = not st.ghostMode
        setGhostMode(st.ghostMode)

    elseif key == "spinBot" then
        st.spinBot = not st.spinBot
        notify(st.spinBot and "🌀 Spin Bot ON" or "❌ Spin Bot OFF", st.spinBot and C.success or C.error)
    elseif key == "toggleGui" then
        st.gui = not st.gui
        if obj.panel then obj.panel.Visible = st.gui end
        if obj.shadow then obj.shadow.Visible = st.gui end
    end
end

-- Panic
local function doPanic()
    for k in pairs(st) do
        if k ~= "running" and k ~= "gui" then st[k] = false end
    end
    rmbDown = false
    obj.lockedTarget = nil
    disableFly()
    resetAllHitboxes()
    setFullbright(false)
    spDisable()
    notify("🚨 PANIC! All disabled", C.error, "🚨")
end

-- Eject
local function doEject()
    print("[Medusa] Ejecting...")
    st.running = false
    doPanic()
    for _, conn in ipairs(obj.connections) do pcall(function() conn:Disconnect() end) end
    clearESP()
    for _, gui in ipairs(SP.allGuis) do pcall(function() gui:Destroy() end) end
    if getgenv then getgenv().MedusaLoaded = false end
    print("[Medusa] ✅ Ejected!")
end

                obj.crosshairParts[2].Position = UDim2.new(0, cx, 0, cy + g)
                obj.crosshairParts[2].Size = UDim2.new(0, 2, 0, s)
            end

                obj.crosshairParts[3].Position = UDim2.new(0, cx - g, 0, cy)
                obj.crosshairParts[3].Size = UDim2.new(0, s, 0, 2)

            if obj.crosshairParts[4] then
                obj.crosshairParts[4].Position = UDim2.new(0, cx + g, 0, cy)
                obj.crosshairParts[4].Size = UDim2.new(0, s, 0, 2)

        end
    end))
end

            ul.TextSize = 10
            ul.TextColor3 = C.textMuted
            ul.TextXAlignment = Enum.TextXAlignment.Left
            ul.Text = "@" .. plr.Name

            hpFill.BackgroundColor3 = C.success
            hpFill.BorderSizePixel = 0
            hpFill.Parent = hpBg
            Instance.new("UICorner", hpFill).CornerRadius = UDim.new(1, 0)

                local hum = plr.Character and plr.Character:FindFirstChildOfClass("Humanoid")
                if hum and hum.MaxHealth > 0 then

                    hpFill.Size = UDim2.new(pct, 0, 1, 0)
                    if pct < 0.3 then hpFill.BackgroundColor3 = C.error

                b.Size = UDim2.new(0, 50, 0, 18)
                b.Position = UDim2.new(1, xPos, 0, 0)
                b.BackgroundColor3 = color

                b.BorderSizePixel = 0
                b.AutoButtonColor = false
                b.Font = Enum.Font.GothamBold

                    if hum then
                        camera.CameraSubject = hum
                        obj.spectating = plr
                        notify("📷 Spectating " .. plr.Name, C.blue)

            end)

                    local tHrp = plr.Character and plr.Character:FindFirstChild("HumanoidRootPart")
                    if myHrp and tHrp then

                        myHrp.CFrame = tHrp.CFrame
                        myHrp.Velocity = Vector3.new(9999, 9999, 9999)

                end)

                    end
                end)
            end)

        end
    end

-- ═══ S31B: FPS/PING UPDATE LOOP ═══
task.spawn(function()
    local frameCount = 0
    local lastTime = os.clock()
    local fps = 60
    local ping = 0

    -- FPS counter via RenderStepped
    addConn(RS.RenderStepped:Connect(function()
        frameCount = frameCount + 1
    end))

    while st.running do
        task.wait(0.5)
        -- Calculate FPS
        local now = os.clock()
        local dt = now - lastTime
        if dt > 0 then
            fps = math.floor(frameCount / dt)
            frameCount = 0
            lastTime = now
        end

        -- Calculate Ping
        pcall(function()
            local stats = game:GetService("Stats")
            local net = stats:FindFirstChild("PerformanceStats")
            if net then
                local pingItem = net:FindFirstChild("Ping")
                if pingItem then ping = math.floor(pingItem:GetValue()) end
            end
        end)
        -- Fallback ping method
        if ping == 0 then
            pcall(function()
                ping = math.floor(player:GetNetworkPing() * 1000)
            end)
        end

        -- Update watermark
        if obj.watermark then
            pcall(function()
                obj.watermark.Text = string.format("🐍 MEDUSA v13 | %d FPS | %dms", fps, ping)
            end)
        end

        -- Update topbar FPS/Ping label
        if obj.fpsPingLabel then
            pcall(function()
                obj.fpsPingLabel.Text = string.format("%d FPS | %dms", fps, ping)
            end)
        end

        -- Status pills are updated in the RenderStepped aimbot loop (S22)

        -- Ghost Mode hover detection
        if st.ghostMode and obj.panel then
            pcall(function()
                local mp = UIS:GetMouseLocation()
                local pp = obj.panel.AbsolutePosition
                local ps = obj.panel.AbsoluteSize
                local isOver = mp.X >= pp.X and mp.X <= pp.X + ps.X and mp.Y >= pp.Y and mp.Y <= pp.Y + ps.Y
                local targetTrans = isOver and cfg.gui.panelOpacity or 0.92
                TS:Create(obj.panel, TweenInfo.new(0.3), {BackgroundTransparency = targetTrans}):Play()
            end)
        end

        -- ESP auto-refresh
        if st.esp then
            local rem = cfg.espRefreshRate - (os.time() - lastESPRefresh)
            if rem <= 0 then
                clearESP()
                task.wait(0.2)
                lastESPRefresh = os.time()
            end
        end
    end
end)
